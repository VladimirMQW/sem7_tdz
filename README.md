#Задача 1: 

Задайте значения M и N. Напишите программу, которая 
выведет все натуральные числа в промежутке от M до N. 
Использовать рекурсию, не использовать циклы.

M = 1; N = 5 -> "1, 2, 3, 4, 5"
M = 4; N = 8 -> "4, 5, 6, 7, 8"
M = 8; N = 4 -> "8, 7, 6, 5, 4"

m = 2, n = 3 -> A(m,n) = 29
